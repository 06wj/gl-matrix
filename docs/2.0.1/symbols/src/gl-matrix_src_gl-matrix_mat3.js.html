 35 /**
 36  * Creates a new identity mat3
 37  *
 38  * @returns {mat3} a new 3x3 matrix
 39  */
 40 mat3.create = function() {
 41     var out = new GLMAT_ARRAY_TYPE(9);
 42     out[0] = 1;
 43     out[1] = 0;
 44     out[2] = 0;
 45     out[3] = 0;
 46     out[4] = 1;
 47     out[5] = 0;
 48     out[6] = 0;
 49     out[7] = 0;
 50     out[8] = 1;
 51     return out;
 52 };

 54 /**
 55  * Creates a new mat3 initialized with values from an existing matrix
 56  *
 57  * @param {mat3} a matrix to clone
 58  * @returns {mat3} a new 3x3 matrix
 59  */
 60 mat3.clone = function(a) {
 61     var out = new GLMAT_ARRAY_TYPE(9);
 62     out[0] = a[0];
 63     out[1] = a[1];
 64     out[2] = a[2];
 65     out[3] = a[3];
 66     out[4] = a[4];
 67     out[5] = a[5];
 68     out[6] = a[6];
 69     out[7] = a[7];
 70     out[8] = a[8];
 71     return out;
 72 };

 74 /**
 75  * Copy the values from one mat3 to another
 76  *
 77  * @param {mat3} out the receiving matrix
 78  * @param {mat3} a the source matrix
 79  * @returns {mat3} out
 80  */
 81 mat3.copy = function(out, a) {
 82     out[0] = a[0];
 83     out[1] = a[1];
 84     out[2] = a[2];
 85     out[3] = a[3];

 94 /**
 95  * Set a mat3 to the identity matrix
 96  *
 97  * @param {mat3} out the receiving matrix
 98  * @returns {mat3} out
 99  */
100 mat3.identity = function(out) {
101     out[0] = 1;
102     out[1] = 0;
103     out[2] = 0;
104     out[3] = 0;
105     out[4] = 1;
106     out[5] = 0;
107     out[6] = 0;
108     out[7] = 0;
109     out[8] = 1;
110     return out;
111 };

113 /**
114  * Transpose the values of a mat3
115  *
116  * @param {mat3} out the receiving matrix
117  * @param {mat3} a the source matrix
118  * @returns {mat3} out
119  */
120 mat3.transpose = function(out, a) {
121     // If we are transposing ourselves we can skip a few steps but have to cache some values
122     if (out === a) {
123         var a01 = a[1], a02 = a[2], a12 = a[5];
124         out[1] = a[3];
125         out[2] = a[6];
126         out[3] = a01;
127         out[5] = a[7];
128         out[6] = a02;
129         out[7] = a12;
130     } else {
131         out[0] = a[0];
132         out[1] = a[3];
133         out[2] = a[6];
134         out[3] = a[1];
135         out[4] = a[4];
136         out[5] = a[7];
137         out[6] = a[2];
138         out[7] = a[5];
139         out[8] = a[8];
140     }
141     
142     return out;
143 };

145 /**
146  * Inverts a mat3
147  *
148  * @param {mat3} out the receiving matrix
149  * @param {mat3} a the source matrix
150  * @returns {mat3} out
151  */
152 mat3.invert = function(out, a) {
153     var a00 = a[0], a01 = a[1], a02 = a[2],
154         a10 = a[3], a11 = a[4], a12 = a[5],
155         a20 = a[6], a21 = a[7], a22 = a[8],
156 
157         b01 = a22 * a11 - a12 * a21,
158         b11 = -a22 * a10 + a12 * a20,
159         b21 = a21 * a10 - a11 * a20,
160 
161         // Calculate the determinant
162         det = a00 * b01 + a01 * b11 + a02 * b21;
163 
164     if (!det) { 
165         return null; 
166     }
167     det = 1.0 / det;
168 
169     out[0] = b01 * det;
170     out[1] = (-a22 * a01 + a02 * a21) * det;
171     out[2] = (a12 * a01 - a02 * a11) * det;
172     out[3] = b11 * det;
173     out[4] = (a22 * a00 - a02 * a20) * det;
174     out[5] = (-a12 * a00 + a02 * a10) * det;
175     out[6] = b21 * det;
176     out[7] = (-a21 * a00 + a01 * a20) * det;
177     out[8] = (a11 * a00 - a01 * a10) * det;
178     return out;
179 };

181 /**
182  * Caclulates the adjugate of a mat3
183  *
184  * @param {mat3} out the receiving matrix
185  * @param {mat3} a the source matrix
186  * @returns {mat3} out
187  */
188 mat3.adjoint = function(out, a) {
189     var a00 = a[0], a01 = a[1], a02 = a[2],
190         a10 = a[3], a11 = a[4], a12 = a[5],
191         a20 = a[6], a21 = a[7], a22 = a[8];
192 
193     out[0] = (a11 * a22 - a12 * a21);
194     out[1] = (a02 * a21 - a01 * a22);
195     out[2] = (a01 * a12 - a02 * a11);
196     out[3] = (a12 * a20 - a10 * a22);
197     out[4] = (a00 * a22 - a02 * a20);
198     out[5] = (a02 * a10 - a00 * a12);
199     out[6] = (a10 * a21 - a11 * a20);
200     out[7] = (a01 * a20 - a00 * a21);
201     out[8] = (a00 * a11 - a01 * a10);
202     return out;
203 };

205 /**
206  * Calculates the determinant of a mat3
207  *
208  * @param {mat3} a the source matrix
209  * @returns {Number} determinant of a
210  */
211 mat3.determinant = function (a) {
212     var a00 = a[0], a01 = a[1], a02 = a[2],
213         a10 = a[3], a11 = a[4], a12 = a[5],
214         a20 = a[6], a21 = a[7], a22 = a[8];
215 
216     return a00 * (a22 * a11 - a12 * a21) + a01 * (-a22 * a10 + a12 * a20) + a02 * (a21 * a10 - a11 * a20);
217 };

219 /**
220  * Multiplies two mat3's
221  *
222  * @param {mat3} out the receiving matrix
223  * @param {mat3} a the first operand
224  * @param {mat3} b the second operand
225  * @returns {mat3} out
226  */
227 mat3.multiply = function (out, a, b) {
228     var a00 = a[0], a01 = a[1], a02 = a[2],
229         a10 = a[3], a11 = a[4], a12 = a[5],
230         a20 = a[6], a21 = a[7], a22 = a[8],
231 
232         b00 = b[0], b01 = b[1], b02 = b[2],
233         b10 = b[3], b11 = b[4], b12 = b[5],
234         b20 = b[6], b21 = b[7], b22 = b[8];
235 
236     out[0] = b00 * a00 + b01 * a10 + b02 * a20;
237     out[1] = b00 * a01 + b01 * a11 + b02 * a21;
238     out[2] = b00 * a02 + b01 * a12 + b02 * a22;
239 
240     out[3] = b10 * a00 + b11 * a10 + b12 * a20;
241     out[4] = b10 * a01 + b11 * a11 + b12 * a21;
242     out[5] = b10 * a02 + b11 * a12 + b12 * a22;
243 
244     out[6] = b20 * a00 + b21 * a10 + b22 * a20;
245     out[7] = b20 * a01 + b21 * a11 + b22 * a21;
246     out[8] = b20 * a02 + b21 * a12 + b22 * a22;
247     return out;
248 };

256 /**
257  * Returns a string representation of a mat3
258  *
259  * @param {mat3} mat matrix to represent as a string
260  * @returns {String} string representation of the matrix
261  */
262 mat3.str = function (a) {
263     return 'mat3(' + a[0] + ', ' + a[1] + ', ' + a[2] + ', ' + 
264                     a[3] + ', ' + a[4] + ', ' + a[5] + ', ' + 
265                     a[6] + ', ' + a[7] + ', ' + a[8] + ')';
266 };
